# a broad measure

# simple average of all buys
b = trades[trades['side']=='B']['lastPx'].mean()
s = trades[trades['side']=='S']['lastPx'].mean()
print(f'the simple average buy price is {round(b*100,2)} %')
print(f'the simple average sell price is {round(s*100,2)} %')

# weighted average of all buys
wb = (trades[trades['side']=='B']['lastPx'] * trades[trades['side']=='B']['lastQty']).sum()
wb = wb / trades[trades['side']=='B']['lastQty'].sum()
print(f'the weighted average buy price is {round(wb*100,2)} %')
ws = (trades[trades['side']=='S']['lastPx'] * trades[trades['side']=='S']['lastQty']).sum()
ws = ws / trades[trades['side']=='S']['lastQty'].sum()
print(f'the weighted average sell price is {round(ws*100,2)} %')
print('---')
print(f'the weighted net gain: {round((ws-wb)*100, 2)} %')
